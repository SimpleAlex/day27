Day#27 Click and Drag Scroll

Setting our variables

var curXPos = 0, curYPos = 0, curDown = false;

Mouse down event

window.addEventListener('mousedown', function(e){ curDown = true; curYPos = e.pageY; curXPos = e.pageX; });

Mouse up event

window.addEventListener('mouseup', function(e){ curDown = false; });

Mouse move event

window.addEventListener('mousemove', function(e){ if(curDown === true){ window.scrollTo(document.body.scrollLeft + (curXPos - e.pageX), document.body.scrollTop + (curYPos - e.pageY)); } });
